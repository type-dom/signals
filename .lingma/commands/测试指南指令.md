# TypeDom Signals Testing Specification

You are the quality assurance expert for TypeDom Signals. You design comprehensive test strategies, ensure edge case coverage, and maintain >90% code coverage across all components.

## Context

- **Test Framework**: Vitest 3.x
- **Coverage Target**: >= 90% statements, functions, branches
- **Test Files**: 12 spec files covering all APIs
- **Performance Tests**: Included in core test suite

## Configuration Defaults

| Setting                   | Default       | Description                                                               |
| ------------------------- | ------------- | ------------------------------------------------------------------------- |
| `--coverage-threshold`    | 90            | Minimum coverage percentage                                               |
| `--test-timeout`          | 5000          | Default test timeout in milliseconds                                      |
| `--parallel`              | true          | Run tests in parallel when possible                                       |
| `--watch`                 | false         | Watch mode for development                                                |

### Test Structure Pattern

```typescript
import { test, expect } from 'vitest';
import { signal, computed, effect } from '../src';

test('should description of behavior', () => {
  // Arrange - Setup test data
  const src = signal(0);
  
  // Act - Execute operation
  src.set(1);
  
  // Assert - Verify result
  expect(src.get()).toBe(1);
});
```

## Core API Tests

### Signal Tests

#### Basic Operations
```typescript
test('should create signal with initial value', () => {
  const count = signal(0);
  expect(count.get()).toBe(0);
});

test('should update signal value', () => {
  const count = signal(0);
  count.set(1);
  expect(count.get()).toBe(1);
});

test('should notify subscribers on change', () => {
  const count = signal(0);
  let notifiedValue: number | undefined;
  
  effect(() => {
    notifiedValue = count.get();
  });
  
  expect(notifiedValue).toBe(0);
  count.set(1);
  expect(notifiedValue).toBe(1);
});
```

#### Reference Type Handling
```typescript
test('should handle object updates immutably', () => {
  const user = signal({ name: 'John', age: 30 });
  
  // Correct: New object
  user.set({ ...user.get(), age: 31 });
  expect(user.get().age).toBe(31);
  
  // Wrong: Direct mutation won't trigger
  // user.get().age = 32;
  // expect(user.get().age).toBe(32); // May fail
});

test('should handle array mutations with trigger', () => {
  const items = signal<number[]>([1, 2, 3]);
  
  items.get().push(4);
  trigger(items);
  
  expect(items.get().length).toBe(4);
});
```

### Computed Tests

#### Dependency Propagation
```typescript
test('should correctly propagate through computed chain', () => {
  const src = signal(0);
  const c1 = computed(() => src.get()! % 2);
  const c2 = computed(() => c1.get());
  const c3 = computed(() => c2.get());

  c3.get();
  src.set(1);
  c2.get();
  src.set(3);

  expect(c3.get()).toBe(1);
});
```

#### Caching Behavior
```typescript
test('should cache computed value until dirty', () => {
  const count = signal(0);
  let computeCount = 0;

  const result = computed(() => {
    computeCount++;
    return count.get() * 2;
  });

  result.get();
  expect(computeCount).toBe(1);

  result.get(); // Cached
  expect(computeCount).toBe(1);

  count.set(1); // Mark dirty
  result.get(); // Recompute
  expect(computeCount).toBe(2);
});
```

#### Value Reversion Optimization
```typescript
test('should not update if value reverted', () => {
  let times = 0;

  const src = signal(0);
  const c1 = computed(() => {
    times++;
    return src.get();
  });
  
  c1.get();
  expect(times).toBe(1);
  
  src.set(1);
  src.set(0); // Revert
  
  c1.get();
  expect(times).toBe(1); // Should NOT recompute
});
```

### Effect Tests

#### Basic Tracking
```typescript
test('should clear subscriptions when stopped', () => {
  let bRunTimes = 0;

  const a = signal(1);
  const b = computed(() => {
    bRunTimes++;
    return a.get()! * 2;
  });
  
  const stopEffect = effect(() => {
    b.get();
  });

  expect(bRunTimes).toBe(1);
  a.set(2);
  expect(bRunTimes).toBe(2);
  
  stopEffect();
  a.set(3);
  expect(bRunTimes).toBe(2); // No longer triggers
});
```

#### Nested Effects
```typescript
test('should not run untracked inner effect', () => {
  const a = signal(3);
  const b = computed(() => a.get()! > 0);

  effect(() => {
    if (b.get()) {
      effect(() => {
        if (a.get() == 0) {
          throw new Error('bad');
        }
      });
    }
  });

  a.set(2);
  a.set(1);
  a.set(0); // Should not throw
});
```

#### Execution Order
```typescript
test('should run outer effect first', () => {
  const a = signal(1);
  const b = signal(1);

  effect(() => {
    if (a.get()) {
      effect(() => {
        b.get();
        if (a.get() == 0) {
          throw new Error('bad');
        }
      });
    }
  });

  startBatch();
  b.set(0);
  a.set(0);
  endBatch(); // Should not throw
});
```

### Batch Operation Tests

```typescript
test('should support batched updates', () => {
  function batchEffect(fn: () => void) {
    return effect(() => {
      startBatch();
      try {
        return fn();
      } finally {
        endBatch();
      }
    });
  }

  const logs: string[] = [];
  const a = signal(0);
  const b = signal(0);

  const aa = computed(() => {
    logs.push('aa-0');
    if (!a.get()) {
      b.set(1);
    }
    logs.push('aa-1');
  });

  const bb = computed(() => {
    logs.push('bb');
    return b.get();
  });

  batchEffect(() => { bb.get(); });
  batchEffect(() => { aa.get(); });

  expect(logs).toEqual(['bb', 'aa-0', 'aa-1', 'bb']);
});
```

### Trigger Tests

```typescript
test('should not throw with no dependencies', () => {
  trigger(() => { });
});

test('should trigger computed updates', () => {
  const arr = signal<number[]>([]);
  const length = computed(() => arr.get().length);

  expect(length.get()).toBe(0);
  arr.get().push(1);
  trigger(arr);
  expect(length.get()).toBe(1);
});

test('should trigger effect once', () => {
  const src1 = signal<number[]>([]);
  const src2 = signal<number[]>([]);

  let triggers = 0;

  effect(() => {
    triggers++;
    src1.get();
    src2.get();
  });

  expect(triggers).toBe(1);
  trigger(() => {
    src1.get();
    src2.get();
  });
  expect(triggers).toBe(2);
});
```

## Advanced Test Scenarios

### Complex Topology Tests

#### Diamond Dependency
```typescript
test('diamond dependency pattern', () => {
  const a = signal(1);
  const b = computed(() => a.get() * 2);
  const c = computed(() => a.get() + 1);
  const d = computed(() => b.get() + c.get());

  let executions = 0;
  effect(() => {
    executions++;
    d.get();
  });

  expect(executions).toBe(1);
  a.set(2);
  expect(executions).toBe(2);
  expect(d.get()).toBe(7); // 2*2 + 2+1
});
```

#### Circular Dependencies
```typescript
test('should handle circular dependencies safely', () => {
  const a = signal(false);
  const b = computed(() => a.get());
  const c = computed(() => {
    b.get();
    return 0;
  });
  const d = computed(() => {
    c.get();
    return b.get();
  });

  let triggers = 0;
  effect(() => {
    d.get();
    triggers++;
  });
  
  expect(triggers).toBe(1);
  a.set(true);
  expect(triggers).toBe(2);
});
```

### Performance Tests

#### Large Scale Updates
```typescript
test('should handle large scale updates efficiently', () => {
  const signals = Array.from({ length: 1000 }, () => signal(0));
  
  const total = computed(() => {
    return signals.reduce((sum, s) => sum + s.get(), 0);
  });

  startBatch();
  signals.forEach((s, i) => s.set(i));
  endBatch();

  expect(total.get()).toBe(499500); // Sum 0..999
});
```

#### Deep Nesting
```typescript
test('should handle deep nesting without stack overflow', () => {
  const base = signal(1);
  let prev = base;
  
  // Create 100 levels of nesting
  for (let i = 0; i < 100; i++) {
    prev = computed(() => prev.get() + 1);
  }

  expect(prev.get()).toBe(101);
  
  base.set(2);
  expect(prev.get()).toBe(102);
});
```

### Memory Leak Tests

#### Effect Cleanup
```typescript
test('should cleanup effect properly', () => {
  const src = signal(0);
  let executions = 0;

  const stop = effect(() => {
    executions++;
    src.get();
  });

  expect(executions).toBe(1);
  src.set(1);
  expect(executions).toBe(2);

  stop();
  src.set(2);
  expect(executions).toBe(2); // No longer executes
});
```

#### Scope Cleanup
```typescript
test('should cleanup effectScope properly', () => {
  const src = signal(0);
  let executions = 0;

  const stopScope = effectScope(() => {
    effect(() => {
      executions++;
      src.get();
    });
    
    effect(() => {
      executions++;
      src.get();
    });
  });

  expect(executions).toBe(2);
  src.set(1);
  expect(executions).toBe(4);

  stopScope();
  src.set(2);
  expect(executions).toBe(4); // No longer executes
});
```

### Best Practice Tests

#### Immutable Pattern
```typescript

  // Correct: Create new object
  user.set({ ...user.get(), age: 31 });
```

#### Array Update Pattern
```typescript
test('should use trigger for array mutations', () => {
  const items = signal<number[]>([1, 2, 3]);
  
  // Method 1: New array (recommended)
  items.set([...items.get(), 4]);
  expect(items.get().length).toBe(4);
  
  // Method 2: Use trigger
  items.get().push(5);
  trigger(items);
  expect(items.get().length).toBe(5);
});
```

### Statement Coverage (>= 90%)
```typescript
test('should cover all branches', () => {
  const val = signal(undefined as number | undefined);
  
  // Test undefined branch
  expect(val.get()).toBeUndefined();
  
  // Test number branch

  expect(val.get()).toBe(42);
});
```

### Branch Coverage (>= 80%)
```typescript
test('should test both conditional branches', () => {
  const cond = signal(true);
  const result = computed(() => cond.get() ? 'yes' : 'no');
  
  expect(result.get()).toBe('yes');
  cond.set(false);
  expect(result.get()).toBe('no');
});
```

### Function Coverage (>= 90%)
```typescript
test('should test all exported functions', () => {
  // signal
  const s = signal(0);
  expect(s.get()).toBe(0);
  
  // computed
  const c = computed(() => s.get());
  expect(c.get()).toBe(0);
  
  // effect
  let executed = false;
  effect(() => { executed = true; s.get(); });
  expect(executed).toBe(true);
  
  // trigger
  trigger(() => { s.get(); });
});
```
