## Architecture Overview

### Core Components

1. **ReactiveNode Interface**: Base interface for all reactive objects
2. **Link Structure**: Bidirectional linked list for dependency tracking  
3. **Signal<T> Class**: Readable/writable reactive signal
4. **Computed<T> Class**: Cached computed property with lazy evaluation
5. **Effect Function**: Side-effect tracker with automatic re-execution
6. **EffectScope**: Lifecycle manager for grouped effects

### Dependency Graph

```
signal.get() → tracks dependency via link()
       ↓
effect/computed → stores in deps chain
       ↓
signal.set() → propagates via propagate()
       ↓
queued effects → flushed via flush()
```

## Coding Standards

### TypeScript Type System

#### Rule 1: Explicit Type Definitions (Mandatory)
```typescript
// ✅ CORRECT: Explicit interface with generics
export interface ComputedNode<T = any, S = T> extends ReactiveNode {
  value: T | undefined;
  getter: (previousValue?: S) => T;
  setter?: (newValue: S) => void;
}

// ❌ WRONG: Using any type
interface BadNode {
  value: any;  // Violation: loses type safety
  fn: any;     // Violation: no type information
}
```

**Enforcement**: Run `tsc --noEmit` before commit, fail build on errors

#### Rule 2: Const Enum for Performance
```typescript
// ✅ CORRECT: const enum for compile-time inlining
export const enum ReactiveFlags {
  None = 0,
  Mutable = 1,
  Watching = 2,
  RecursedCheck = 4,
  Recursed = 8,
  Dirty = 16,
  Pending = 32,
}

// ❌ WRONG: Regular enum creates runtime object
enum OldFlags {
  None = 0,
  Mutable = 1,  // Violation: Creates runtime object, slower
}
```

**Performance Impact**: const enum eliminates property access overhead (~30% faster)

#### Rule 3: Bitwise Flag Operations
```typescript
// ✅ CORRECT: Bitwise operations for flag combinations
node.flags = ReactiveFlags.Mutable | ReactiveFlags.Dirty;

if (flags & ReactiveFlags.Watching) {
  notify(sub);
}

// ❌ WRONG: Multiple boolean fields
interface Node {
  isMutable: boolean;   // Wastes memory, slower checks
  isDirty: boolean;     // Multiple memory accesses
  isWatching: boolean;  // No compile-time optimization
}
```

**Memory Savings**: 1 number vs 3+ booleans, single bitwise check vs multiple conditionals

## Performance Optimization Rules

### Rule 1: No Recursion (Critical)
```typescript
// ✅ CORRECT: Stack-based iteration
function propagate(link: Link): void {
  let next = link.nextSub;
  let stack: Stack<Link | undefined> | undefined;

  top: do {
    const sub = link.sub;
    // ... processing logic
    if (subSubs !== undefined) {
      link = subSubs;
      if (subSubs.nextSub !== undefined) {
        stack = { value: next, prev: stack };
        next = link.nextSub;
      }
      continue;
    }
    // Backtracking logic
  } while (true);
}

// ❌ WRONG: Recursive calls cause stack overflow
function propagate(link: Link): void {
  const subSubs = sub.subs;
  if (subSubs !== undefined) {
    propagate(subSubs);  // Violation: recursion depth limit
  }
}
```

**Performance Impact**: Supports 1000+ nested computeds vs ~150 with recursion

### Rule 2: Lazy Evaluation (Mandatory for Computed)
```typescript
// ✅ CORRECT: Re-compute only when dirty
get() {
  const flags = this.flags;
  if (flags & ReactiveFlags.Dirty || flags & ReactiveFlags.Pending) {
    if (updateComputed(this)) {
      shallowPropagate(this.subs);
    }
  }
  return this.value;
}

// ❌ WRONG: Always re-computes
get() {
  this.value = this.getter();  // Violation: no caching
  return this.value;
}
```

**Benchmark**: 100x improvement for frequently-accessed computeds

### Rule 3: Batch Updates (Required for Multiple Changes)
```typescript
// ✅ CORRECT: Batch multiple updates
startBatch();
count.set(1);
double.set(2);
endBatch(); // Single propagation

// ❌ WRONG: Individual triggers
count.set(1); // Triggers full propagation
double.set(2); // Triggers another full propagation
```

**Performance Impact**: O(n) vs O(n²) for n signals

### Rule 4: Dependency Tracking Optimization (O(1) Required)
```typescript
// ✅ CORRECT: Bidirectional linked list, O(1) operations
function link(dep: ReactiveNode, sub: ReactiveNode, version: number): void {
  const prevDep = sub.depsTail;
  if (prevDep !== undefined && prevDep.dep === dep) {
    return; // O(1) fast path
  }
  // ... linking logic
}

// ❌ WRONG: Array lookup O(n)
function addDep(deps: Dep[], dep: Dep) {
  if (!deps.includes(dep)) {  // Violation: O(n) lookup
    deps.push(dep);
  }
}
```

**Scalability**: Supports 10,000+ dependencies vs ~500 with arrays

### Rule 5: V8 Fast Path Optimization
```typescript
// ✅ CORRECT: Keep class properties under 10
class Signal<T> {
  pendingValue: T | undefined;  // 1
  currentValue: T | undefined;  // 2
  subs?: Link;                  // 3
  subsTail?: Link;              // 4
  flags: ReactiveFlags;         // 5
  // Total: 5 properties - within fast path limit
}

// ❌ WRONG: Too many properties
class BloatedSignal {
  // 15+ properties  // Violation: exceeds V8 fast path limit
}
```

**Performance Impact**: 2-3x faster property access within fast path

## Error Handling Standards

### Exception Safety (Critical)
```typescript
// ✅ CORRECT: try-finally ensures state restoration
export function effect(fn: () => void): () => void {
  const e: EffectNode = { /* ... */ };
  const prevSub = setActiveSub(e);
  try {
    e.fn();
  } finally {
    activeSub = prevSub; // Ensure restoration
    e.flags &= ~ReactiveFlags.RecursedCheck;
  }
  return effectOper.bind(e);
}

// ❌ WRONG: State not restored on error
export function effect(fn: () => void): () => void {
  const e: EffectNode = { /* ... */ };
  setActiveSub(e);
  e.fn(); // Violation: if throws, state corrupted
  return effectOper.bind(e);
}
```

### Flush Error Recovery
```typescript
// ✅ CORRECT: Queue cleanup even on error
function flush(): void {
  try {
    while (notifyIndex < queuedLength) {
      const effect = queued[notifyIndex]!;
      queued[notifyIndex++] = undefined;
      run(effect);
    }
  } finally {
    // Cleanup remaining queue
    while (notifyIndex < queuedLength) {
      const effect = queued[notifyIndex]!;
      queued[notifyIndex++] = undefined;
      effect.flags |= ReactiveFlags.Watching | ReactiveFlags.Recursed;
    }
    notifyIndex = 0;
    queuedLength = 0;
  }
}
```

## Testing Standards

### Unit Test Structure (AAA Pattern)
```typescript
import { test, expect } from 'vitest';
import { signal, computed, effect } from '../src';

test('should correctly propagate computed changes', () => {
  // Arrange - Setup data
  const src = signal(0);
  const c1 = computed(() => src.get()! % 2);
  
  // Act - Execute operation
  c1.get();
  src.set(1);
  
  // Assert - Verify result
  expect(c1.get()).toBe(1);
});
```

### Boundary Condition Coverage
```typescript
test('should not update if value reverted', () => {
  let times = 0;

  const src = signal(0);
  const c1 = computed(() => {
    times++;
    return src.get();
  });
  
  c1.get();
  expect(times).toBe(1);
  
  src.set(1);
  src.set(0); // Revert to original
  
  c1.get();
  expect(times).toBe(1); // Should NOT re-compute
});
```

### Complex Topology Tests
```typescript
test('should handle side effects with inner effects', () => {
  const a = signal(0);
  const b = signal(0);
  const order: string[] = [];

  effect(() => {
    effect(() => {
      a.get();
      order.push('a');
    });
    effect(() => {
      b.get();
      order.push('b');
    });
    expect(order).toEqual(['a', 'b']);

    order.length = 0;
    b.set(1);
    a.set(1);
    expect(order).toEqual(['b', 'a']);
  });
});
```
