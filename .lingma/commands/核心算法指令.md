# TypeDom Signals Core Algorithm Specification

You are the algorithm expert for TypeDom Signals' reactive system. You understand and can explain the Push-Pull model, dependency tracking, propagation mechanisms, and optimization strategies at a deep technical level.

## Context

- **Algorithm**: Push-Pull hybrid model
- **Inspiration**: Alien Signals with optimizations

## Architecture Overview

### Reactive System Components

```
┌─────────────────────────────────────────────────────┐
│              ReactiveNode Interface                 │
│  deps?: Link | depsTail?: Link | subs?: Link       │
│  subsTail?: Link | flags: ReactiveFlags            │
└─────────────────────────────────────────────────────┘
                         ↓
┌─────────────────────────────────────────────────────┐
│          Bidirectional Link Structure               │
│  version | dep | sub | prev/next pointers           │
│  Enables O(1) link/unlink operations                │
└─────────────────────────────────────────────────────┘
                         ↓
┌─────────────────────────────────────────────────────┐
│         createReactiveSystem Factory                │
│  Returns: link, unlink, propagate, checkDirty       │
│  Configurable via update/notify/unwatched callbacks │
└─────────────────────────────────────────────────────┘
```

### Dependency Graph Flow

```
signal.set(value)
    ↓
[Push Phase] Mark Dirty + Mutable
    ↓
propagate(subs) - Stack-based iteration
    ↓
Mark subscribers as Pending
    ↓
notify(effects) - Queue for flush
    ↓
[Pull Phase] When computed.get() called
    ↓
checkDirty(deps) - Verify if recomputation needed
    ↓
update() only if actually dirty
    ↓
shallowPropagate() to dependents
```

## Dependency Tracking Mechanism

### Link Structure Design

```typescript
interface Link {
  version: number;        // Cycle counter for deduplication
  dep: ReactiveNode;      // Dependency node
  sub: ReactiveNode;      // Subscriber node
  prevSub: Link | undefined;
  nextSub: Link | undefined;
  prevDep: Link | undefined;
  nextDep: Link | undefined;
}
```

**Design Rationale**:
- `version`: Prevents duplicate dependencies in same cycle
- Bidirectional: Enables O(1) removal from both ends
- Doubly-linked: No array traversal needed

### link() Function - O(1) Optimization

```typescript
function link(dep: ReactiveNode, sub: ReactiveNode, version: number): void {
  const prevDep = sub.depsTail;
  
  // Fast path: Already last dependency
  if (prevDep !== undefined && prevDep.dep === dep) {
    return;
  }
  
  // Check if in dependency chain
  const nextDep = prevDep !== undefined ? prevDep.nextDep : sub.deps;
  if (nextDep !== undefined && nextDep.dep === dep) {
    nextDep.version = version;  // Update version
    sub.depsTail = nextDep;     // Move to tail
    return;
  }
  
  // Check if in subscriber chain (same cycle)
  const prevSub = dep.subsTail;
  if (prevSub !== undefined && 
      prevSub.version === version && 
      prevSub.sub === sub) {
    return;
  }
  
  // Create new Link node
  const newLink = sub.depsTail = dep.subsTail = {
    version,
    dep,
    sub,
    prevDep,
    nextDep,
    prevSub,
    nextSub: undefined,
  };
  
  // Update neighbors
  if (nextDep !== undefined) nextDep.prevDep = newLink;
  if (prevDep !== undefined) prevDep.nextDep = newLink;
  if (prevSub !== undefined) prevSub.nextSub = newLink;
  
  // Set as head if empty
  if (!prevDep) sub.deps = newLink;
  if (!prevSub) dep.subs = newLink;
}
```

**Complexity**: O(1) amortized due to tail pointer optimization

### unlink() Function - Clean Removal

```typescript
function unlink(link: Link, sub = link.sub): Link | undefined {
  const dep = link.dep;
  const prevDep = link.prevDep;
  const nextDep = link.nextDep;
  const nextSub = link.nextSub;
  const prevSub = link.prevSub;
  
  // Remove from dep chain
  if (nextDep !== undefined) {
    nextDep.prevDep = prevDep;
  } else {
    sub.depsTail = prevDep;
  }
  
  if (prevDep !== undefined) {
    prevDep.nextDep = nextDep;
  } else {
    sub.deps = nextDep;
  }
  
  // Remove from sub chain
  if (nextSub !== undefined) {
    nextSub.prevSub = prevSub;
  } else {
    dep.subsTail = prevSub;
  }
  
  if (prevSub !== undefined) {
    prevSub.nextSub = nextSub;
  } else if ((dep.subs = nextSub) === undefined) {
    unwatched(dep);  // No more subscribers
  }
  
  return nextDep;
}
```

**Critical**: Calls `unwatched()` callback when last subscriber removed

## Propagation Algorithm

### propagate() - Non-Recursive Implementation

```typescript
function propagate(link: Link): void {
  let next = link.nextSub;
  let stack: Stack<Link | undefined> | undefined;

  top: do {
    const sub = link.sub;
    let flags = sub.flags;

    // State machine for flag transitions
    if (!(flags & (RecursedCheck | Recursed | Dirty | Pending))) {
      sub.flags = flags | Pending;
    } else if (!(flags & (RecursedCheck | Recursed))) {
      flags = None;
    } else if (!(flags & RecursedCheck)) {
      sub.flags = (flags & ~Recursed) | Pending;
    } else if (!(flags & (Dirty | Pending)) && isValidLink(link, sub)) {
      sub.flags = flags | (Recursed | Pending);
      flags &= Mutable;
    } else {
      flags = None;
    }

    // Notify watchers
    if (flags & Watching) {
      notify(sub);
    }

    // Continue propagation down mutable path
    if (flags & Mutable) {
      const subSubs = sub.subs;
      if (subSubs !== undefined) {
        const nextSub = (link = subSubs).nextSub;
        if (nextSub !== undefined) {
          stack = { value: next, prev: stack };
          next = nextSub;
        }
        continue;
      }
    }

    // Backtrack logic
    if ((link = next!) !== undefined) {
      next = link.nextSub;
      continue;
    }

    while (stack !== undefined) {
      link = stack.value!;
      stack = stack.prev;
      if (link !== undefined) {
        next = link.nextSub;
        continue top;
      }
    }

    break;
  } while (true);
}
```

**Key Innovations**:
1. **Stack-based**: Eliminates recursion, prevents stack overflow
2. **State machine**: Complex flag transitions handled correctly
3. **Backtracking**: Properly resumes from saved state

### shallowPropagate() - Lightweight Propagation

```typescript
function shallowPropagate(link: Link): void {
  do {
    const sub = link.sub;
    const flags = sub.flags;
    
    // Only mark Pending → Dirty
    if ((flags & (Pending | Dirty)) === Pending) {
      sub.flags = flags | Dirty;
      
      // Notify if watching
      if ((flags & (Watching | RecursedCheck)) === Watching) {
        notify(sub);
      }
    }
  } while ((link = link.nextSub!) !== undefined);
}
```

**Use Case**: Quick invalidation without full propagation logic

## Dirty Checking Algorithm

### checkDirty() - Lazy Evaluation Control

```typescript
function checkDirty(link: Link, sub: ReactiveNode): boolean {
  let stack: Stack<Link> | undefined;
  let checkDepth = 0;
  let dirty = false;

  top: do {
    const dep = link.dep;
    const flags = dep.flags;

    // Case 1: Current node already dirty
    if (sub.flags & Dirty) {
      dirty = true;
    } 
    // Case 2: Dep is Mutable+Dirty, needs update
    else if ((flags & (Mutable | Dirty)) === (Mutable | Dirty)) {
      if (update(dep)) {
        const subs = dep.subs!;
        if (subs.nextSub !== undefined) {
          shallowPropagate(subs);
        }
        dirty = true;
      }
    } 
    // Case 3: Dep is Mutable+Pending, recurse
    else if ((flags & (Mutable | Pending)) === (Mutable | Pending)) {
      if (link.nextSub !== undefined || link.prevSub !== undefined) {
        stack = { value: link, prev: stack };
      }
      link = dep.deps!;
      sub = dep;
      ++checkDepth;
      continue;
    }

    // Continue checking next dependency
    if (!dirty) {
      const nextDep = link.nextDep;
      if (nextDep !== undefined) {
        link = nextDep;
        continue;
      }
    }

    // Backtrack through check stack
    while (checkDepth--) {
      const firstSub = sub.subs!;
      const hasMultipleSubs = firstSub.nextSub !== undefined;
      
      if (hasMultipleSubs) {
        link = stack!.value;
        stack = stack!.prev;
      } else {
        link = firstSub;
      }
      
      if (dirty) {
        if (update(sub)) {
          if (hasMultipleSubs) {
            shallowPropagate(firstSub);
          }
          sub = link.sub;
          continue;
        }
        dirty = false;
      } else {
        sub.flags &= ~Pending;
      }
      
      sub = link.sub;
      const nextDep = link.nextDep;
      if (nextDep !== undefined) {
        link = nextDep;
        continue top;
      }
    }

    return dirty;
  } while (true);
}
```

**Optimization**: Only recomputes when actually necessary

## Update Execution

### updateComputed() - Lazy Recomputation

```typescript
function updateComputed(c: ComputedNode): boolean {
  ++cycle;  // Increment cycle counter
  c.depsTail = undefined;
  c.flags = Mutable | RecursedCheck;
  
  const prevSub = setActiveSub(c);
  try {
    const oldValue = c.value;
    
    // Special handling for objects (preserve reference for notify)
    if (isObject(oldValue)) {
      c.value = c.getter(oldValue);
      return true;  // Always mark changed for objects
    }
    
    // Strict equality for primitives
    return oldValue !== (c.value = c.getter(oldValue));
  } finally {
    activeSub = prevSub;
    c.flags &= ~RecursedCheck;
    purgeDeps(c);  // Clean up stale dependencies
  }
}
```

## Batch Flush Mechanism

### flush() - Queue Processing

```typescript
function flush(): void {
  try {
    // Process queued effects in order
    while (notifyIndex < queuedLength) {
      const effect = queued[notifyIndex]!;
      queued[notifyIndex++] = undefined;
      run(effect);
    }
  } finally {
    // Cleanup remaining queue (even on error)
    while (notifyIndex < queuedLength) {
      const effect = queued[notifyIndex]!;
      queued[notifyIndex++] = undefined;
      effect.flags |= Watching | Recursed;
    }
    notifyIndex = 0;
    queuedLength = 0;
  }
}
```

### run() - Effect Execution

```typescript
function run(e: EffectNode): void {
  const flags = e.flags;
  
  // Execute if dirty or pending+verified
  if (
    flags & Dirty ||
    (flags & Pending && checkDirty(e.deps!, e))
  ) {
    ++cycle;
    e.depsTail = undefined;
    e.flags = Watching | RecursedCheck;
    
    const prevSub = setActiveSub(e);
    try {
      (e as EffectNode).fn();
    } finally {
      activeSub = prevSub;
      e.flags &= ~RecursedCheck;
      purgeDeps(e);
    }
  } else {
    e.flags = Watching;
  }
}
```

## Notification System

### notify() - Queue Management

```typescript
notify(effect: EffectNode) {
  let insertIndex = queuedLength;
  let firstInsertedIndex = insertIndex;

  do {
    queued[insertIndex++] = effect;
    effect.flags &= ~Watching;
    effect = effect.subs?.sub as EffectNode;
    if (effect === undefined || !(effect.flags & Watching)) {
      break;
    }
  } while (true);

  queuedLength = insertIndex;

  // Reverse order (LIFO) for correct execution
  while (firstInsertedIndex < --insertIndex) {
    const left = queued[firstInsertedIndex];
    queued[firstInsertedIndex++] = queued[insertIndex];
    queued[insertIndex] = left;
  }
}
```

**Rationale**: LIFO ordering ensures inner effects run first

## Cleanup Mechanism

### purgeDeps() - Dependency Cleanup

```typescript
function purgeDeps(sub: ReactiveNode) {
  const depsTail = sub.depsTail;
  let dep = depsTail !== undefined ? depsTail.nextDep : sub.deps;
  
  while (dep !== undefined) {
    dep = unlink(dep, sub);
  }
}
```

### effectScopeOper() - Scope Cleanup

```typescript
function effectScopeOper(this: ReactiveNode): void {
  this.depsTail = undefined;
  this.flags = ReactiveFlags.None;
  purgeDeps(this);
  
  const sub = this.subs;
  if (sub !== undefined) {
    unlink(sub);
  }
}
```

### 4. Lazy Evaluation
```typescript

    updateComputed(this);
```
