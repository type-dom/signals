## 📖 概述

本文档说明 `@type-dom/signals` 项目的监控告警体系，包括性能监控、错误追踪、告警配置等。

### 监控目标

- ✅ **性能指标** - Bundle 大小、执行时间、内存占用
- ✅ **错误追踪** - 运行时错误、类型错误、边界情况
- ✅ **使用统计** - 下载量、活跃度、用户反馈
- ✅ **质量保障** - 测试覆盖率、CI 成功率

### 运行时性能监控

#### 基准测试

```typescript
// tests/benchmarks/performance.bench.ts
import { bench, describe } from 'vitest';
import { signal, computed, effect } from '../../src';

describe('Performance Benchmarks', () => {
  bench('signal creation', () => {
    const sig = signal(0);
  }, {
    iterations: 1000000,
  });
  
  bench('signal get/set cycle', () => {
    const sig = signal(0);
    sig.set(sig.get() + 1);
  }, {
    iterations: 1000000,
  });
  
  bench('computed chain (depth 3)', () => {
    const a = signal(0);
    const b = computed(() => a.get() + 1);
    const c = computed(() => b.get() + 1);
    const d = computed(() => c.get() + 1);
    d.get();
  }, {
    iterations: 100000,
  });
  
  bench('effect execution', () => {
    const sig = signal(0);
    let count = 0;
    
    effect(() => {
      count++;
      sig.get();
    });
    
    sig.set(sig.get() + 1);
  }, {
    iterations: 100000,
  });
});
```

#### 性能指标收集

```typescript
// lib/performance-monitor.ts
interface PerformanceMetrics {
  signalCreation: number[];
  signalGet: number[];
  signalSet: number[];
  computedExecution: number[];
  effectExecution: number[];
  memoryUsage: number[];
}

class PerformanceMonitor {
  private metrics: PerformanceMetrics = {
    signalCreation: [],
    signalGet: [],
    signalSet: [],
    computedExecution: [],
    effectExecution: [],
    memoryUsage: [],
  };
  
  measure<T>(name: keyof PerformanceMetrics, fn: () => T): T {
    const start = performance.now();
    const result = fn();
    const end = performance.now();
    
    this.metrics[name].push(end - start);
    
    // 定期记录内存使用
    if (typeof process !== 'undefined' && process.memoryUsage) {
      this.metrics.memoryUsage.push(process.memoryUsage().heapUsed);
    }
    
    return result;
  }
  
  report(): void {
    console.log('=== Performance Report ===');
    
    Object.entries(this.metrics).forEach(([name, values]) => {
      if (values.length === 0) return;
      
      const avg = values.reduce((a, b) => a + b, 0) / values.length;
      const min = Math.min(...values);
      const max = Math.max(...values);
      
      console.log(`${name}:`);
      console.log(`  Avg: ${avg.toFixed(3)}ms`);
      console.log(`  Min: ${min.toFixed(3)}ms`);
      console.log(`  Max: ${max.toFixed(3)}ms`);
      console.log(`  Count: ${values.length}`);
    });
  }
  
  reset(): void {
    Object.keys(this.metrics).forEach(key => {
      this.metrics[key as keyof PerformanceMetrics] = [];
    });
  }
}

export const perfMonitor = new PerformanceMonitor();
```

## 🚨 错误监控

### Sentry 集成

#### 安装配置

```bash
npm install @sentry/browser @sentry/tracing
```

```typescript
// lib/error-monitoring.ts
import * as Sentry from '@sentry/browser';

Sentry.init({
  dsn: 'https://your-dsn@sentry.io/project-id',
  environment: process.env.NODE_ENV || 'development',
  
  // 性能监控
  tracesSampleRate: 0.2, // 20% 采样率
  
  // 错误过滤
  beforeSend(event, hint) {
    // 忽略某些类型的错误
    if (event.exception) {
      const error = hint.originalException as Error;
      if (error?.message?.includes('Network request failed')) {
        return null;
      }
    }
    return event;
  },
  
  // 标签
  defaultIntegrations: true,
  release: `@type-dom/signals@${VERSION}`,
});
```

#### 错误上报

```typescript
// src/signal.ts
import { perfMonitor } from '../lib/performance-monitor';
import * as Sentry from '@sentry/browser';

export class Signal<T> {
  get(): T {
    try {
      return perfMonitor.measure('signalGet', () => {
        // ... get implementation
      });
    } catch (error) {
      // 上报错误
      Sentry.captureException(error, {
        tags: {
          section: 'signals',
          operation: 'get',
        },
        extra: {
          currentValue: this.currentValue,
          flags: this.flags,
        },
      });
      
      throw error;
    }
  }
  
  set(value: T): void {
    try {
      perfMonitor.measure('signalSet', () => {
        // ... set implementation
      });
    } catch (error) {
      Sentry.captureException(error, {
        tags: {
          section: 'signals',
          operation: 'set',
        },
        extra: {
          value,
          flags: this.flags,
        },
      });
      
      throw error;
    }
  }
}
```

### 自定义错误日志

```typescript
// lib/logger.ts
type LogLevel = 'debug' | 'info' | 'warn' | 'error';

interface LogEntry {
  timestamp: string;
  level: LogLevel;
  message: string;
  context?: any;
  error?: Error;
}

class Logger {
  private logs: LogEntry[] = [];
  private maxLogs = 1000;
  
  log(level: LogLevel, message: string, context?: any, error?: Error): void {
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      context,
      error,
    };
    
    this.logs.push(entry);
    
    // 保持日志数量
    if (this.logs.length > this.maxLogs) {
      this.logs.shift();
    }
    
    // 开发环境输出到控制台
    if (process.env.NODE_ENV === 'development') {
      this.consoleLog(entry);
    }
  }
  
  debug(message: string, context?: any): void {
    this.log('debug', message, context);
  }
  
  info(message: string, context?: any): void {
    this.log('info', message, context);
  }
  
  warn(message: string, context?: any): void {
    this.log('warn', message, context);
  }
  
  error(message: string, context?: any, error?: Error): void {
    this.log('error', message, context, error);
  }
  
  private consoleLog(entry: LogEntry): void {
    const prefix = `[${entry.timestamp}] [${entry.level.toUpperCase()}]`;
    
    switch (entry.level) {
      case 'debug':
        console.debug(prefix, entry.message, entry.context);
        break;
      case 'info':
        console.info(prefix, entry.message, entry.context);
        break;
      case 'warn':
        console.warn(prefix, entry.message, entry.context);
        break;
      case 'error':
        console.error(prefix, entry.message, entry.context, entry.error);
        break;
    }
  }
  
  getLogs(): LogEntry[] {
    return [...this.logs];
  }
  
  clear(): void {
    this.logs = [];
  }
}

export const logger = new Logger();
```

## 📊 告警配置

### 告警规则

```yaml
# .github/alerts.yml
alerts:
  # Bundle 大小告警
  - name: Bundle Size Exceeded
    type: bundlewatch
    condition: "> 10KB"
    file: "dist/signals.umd.min.js"
    notify:
      - github-pr
      - slack
    
  # 测试覆盖率下降
  - name: Coverage Drop
    type: vitest
    condition: "< 90%"
    metric: global.branches
    notify:
      - github-pr
    
  # CI 失败
  - name: CI Failure
    type: github-actions
    workflow: ci.yml
    branches: [main]
    notify:
      - email
      - slack
    
  # 性能回归
  - name: Performance Regression
    type: benchmark
    condition: "> 10%"
    metric: signal_get_avg
    notify:
      - github-issue
```

## 🔍 健康检查

### 定期检查脚本

```bash
#!/bin/bash
# scripts/health-check.sh

echo "=== Health Check ==="

# 1. 检查构建
echo "Building..."
npm run build
if [ $? -ne 0 ]; then
  echo "❌ Build failed"
  exit 1
fi
echo "✅ Build successful"

# 2. 运行测试
echo "Testing..."
npm test
if [ $? -ne 0 ]; then
  echo "❌ Tests failed"
  exit 1
fi
echo "✅ Tests passed"

# 3. 检查覆盖率
echo "Checking coverage..."
npm run test:coverage -- --threshold=90
if [ $? -ne 0 ]; then
  echo "❌ Coverage below threshold"
  exit 1
fi
echo "✅ Coverage OK"

# 4. 检查 Bundle 大小
echo "Checking bundle size..."
npx bundlewatch
if [ $? -ne 0 ]; then
  echo "❌ Bundle size exceeded"
  exit 1
fi
echo "✅ Bundle size OK"

echo "=== All checks passed ==="
```
