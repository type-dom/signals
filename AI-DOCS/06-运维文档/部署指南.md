## 🎯 构建流程

### 环境准备

```bash
# Node.js 版本要求
node >= 18.0.0

# 安装依赖
npm install

# 验证安装
npm run build
```

### 构建配置

```typescript
// rollup.config.ts
import typescript from '@rollup/plugin-typescript';
import resolve from '@rollup/plugin-node-resolve';
import commonjs from '@rollup/plugin-commonjs';
import terser from '@rollup/plugin-terser';
import { defineConfig } from 'rollup';

export default defineConfig({
  input: 'src/index.ts',
  output: [
    // ESM 格式
    {
      file: 'dist/signals.esm.js',
      format: 'esm',
      sourcemap: true,
      exports: 'named',
    },
    // CJS 格式
    {
      file: 'dist/signals.cjs.js',
      format: 'cjs',
      sourcemap: true,
      exports: 'named',
    },
    // UMD 格式（浏览器直接使用）
    {
      file: 'dist/signals.umd.js',
      format: 'umd',
      name: 'TypeDomSignals',
      sourcemap: true,
      exports: 'named',
    },
    // UMD 压缩版
    {
      file: 'dist/signals.umd.min.js',
      format: 'umd',
      name: 'TypeDomSignals',
      sourcemap: true,
      plugins: [terser()],
    },
  ],
  plugins: [
    resolve(), // 解析第三方模块
    commonjs(), // CommonJS 转换
    typescript({
      tsconfig: './tsconfig.json',
      declaration: true,
      declarationDir: './dist',
    }),
  ],
  external: [], // 无外部依赖
  watch: {
    include: 'src/**',
    exclude: 'node_modules/**',
  },
});
```

### 构建脚本

```json
{
  "scripts": {
    "build": "rollup -c",
    "build:watch": "rollup -c -w",
    "prebuild": "npm run clean",
    "clean": "rimraf dist/*",
    "test": "vitest run",
    "test:coverage": "vitest --coverage",
    "prepublishOnly": "npm run test && npm run build"
  }
}
```

### 执行构建

```bash
# 开发构建（监听模式）
npm run build:watch

# 生产构建
npm run build

# 完整流程（清理 → 测试 → 构建）
npm run build
```

### 自动发布（GitHub Actions）

```yaml

  push:
    tags:
      - 'v*'

jobs:
  release:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          registry-url: 'https://registry.npmjs.org'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run tests
        run: npm test
      
      - name: Build
        run: npm run build
      
      - name: Publish to NPM
        run: npm publish --access public
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}
      
      - name: Create GitHub Release
        uses: softprops/action-gh-release@v1
        with:
          generate_release_notes: true
```

## 🔧 持续集成

### GitHub Actions 完整配置

```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
      - run: npm ci
      - run: npm run lint

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
      - run: npm ci
      - run: npm test
      - uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info

  build:
    runs-on: ubuntu-latest
    needs: [lint, test]
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
      - run: npm ci
      - run: npm run build
      - uses: actions/upload-artifact@v3
        with:
          name: dist
          path: dist/

  deploy-cdn:
    runs-on: ubuntu-latest
    needs: build
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/download-artifact@v3
        with:
          name: dist
          path: dist/
      - uses: peaceiris/actions-gh-pages@v3
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./dist
```

## 📊 版本管理

### 语义化版本

```
主版本号。次版本号。修订号
  ↑       ↑       ↑
  Major   Minor   Patch
  
例如：0.8.1
- 0: 主版本号（重大变更，可能不兼容）
- 8: 次版本号（新功能，向后兼容）
- 1: 修订号（Bug 修复，向后兼容）
```

### 版本发布策略

```bash
# 修订号（Bug 修复）
npm version patch  # 0.8.1 → 0.8.2

# 次版本号（新功能）
npm version minor  # 0.8.1 → 0.9.0

# 主版本号（破坏性变更）
npm version major  # 0.8.1 → 1.0.0

# 预发布版本
npm version prerelease  # 0.8.1 → 0.8.2-beta.0
```

## 🔄 回滚策略

### NPM 回滚

```bash
# 取消发布最新版本（24 小时内）
npm unpublish @type-dom/signals@0.8.1

# 恢复到上一个版本
npm publish @type-dom/signals@0.8.0 --tag latest

# 标记某个版本为最新
npm dist-tag add @type-dom/signals@0.8.0 latest
```

### Git 回滚

```bash
# 回滚 commit
git revert HEAD

# 回滚 tag
git tag -d v0.8.1
git push origin :refs/tags/v0.8.1

# 恢复上一版本
git checkout v0.8.0
git tag v0.8.1-hotfix
git push origin v0.8.1-hotfix
```
