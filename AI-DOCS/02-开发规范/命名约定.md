# TypeDOM Signals - 命名约定

> 🏷️ **统一的命名规范和约定**  
> ✅ 提升代码可读性和一致性

---

## 📖 概述

本文档定义了 `@type-dom/signals` 的命名规范，确保代码风格一致、易于理解。

### 命名原则

1. **语义清晰** - 名称应准确表达意图
2. **简洁明了** - 避免冗长但保持清晰
3. **一致性** - 同类概念使用相同命名模式
4. **可搜索** - 避免单字母和模糊缩写

## 🎯 基础命名规则

### 变量命名

```typescript
// ✅ 推荐：使用 camelCase（小驼峰）
const count = signal(0);
const userName = signal('John');
const isActive = signal(true);

// ❌ 不推荐
const Count = signal(0);           // 不应 PascalCase
const user_name = signal('John');  // 不应 snake_case
const c = signal(0);               // 过于简略
```

### 常量和枚举

```typescript
// ✅ 推荐：使用 UPPER_SNAKE_CASE
const MAX_RETRY_COUNT = 3;
const DEFAULT_TIMEOUT = 5000;

export const enum ReactiveFlags {
  NONE = 0,
  MUTABLE = 1,
  WATCHING = 2,
}

// ❌ 不推荐
const maxRetryCount = 3;  // 常量应大写
const max_retry_count = 3; // 混合风格
```

### 类和接口

```typescript
// ✅ 推荐：使用 PascalCase（大驼峰）
export class Signal<T> { }
export interface ReactiveNode { }
export type ComputedGetter<T> = () => T;

// ❌ 不推荐
export class signal<T> { }      // 类名应大写
export interface reactiveNode { } // 接口名应大写
```

### 函数和方法

```typescript
// ✅ 推荐：使用动词 + 名词的 camelCase 形式
function createSignal<T>(value: T): Signal<T> { }
function updateEffect(effect: EffectNode): void { }
function checkDirty(node: ReactiveNode): boolean { }

// ❌ 不推荐
function CreateSignal<T>() { }     // 不应 PascalCase
function update_effect() { }       // 不应 snake_case
function sig() { }                 // 名称不完整
```

## 🔧 特定场景命名

### Signal 相关

```typescript
// ✅ 推荐
const count = signal(0);
const user = signal<User | null>(null);
const items = signal<Item[]>([]);
const isLoading = signal(false);

// 描述性命名
const firstName = signal('John');
const lastName = signal('Doe');
const fullName = computed(() => `${firstName.get()} ${lastName.get()}`);

// ❌ 不推荐
const s = signal(0);              // 无意义
const sig = signal(0);            // 缩写不清晰
const data = signal<any>();       // any 类型
```

### Computed 相关

```typescript
// ✅ 推荐：使用名词或形容词短语
const doubled = computed(() => count.get() * 2);
const total = computed(() => items.get().reduce((sum, item) => sum + item.price, 0));
const isValid = computed(() => email.get().includes('@'));
const fullName = computed(() => `${firstName.get()} ${lastName.get()}`);

// ❌ 不推荐
const comp = computed(() => ...);        // 缩写
const calc = computed(() => ...);        // 不完整
const computeValue = computed(() => ...); // 动词开头，应为名词
```

### Effect 相关

```typescript
// ✅ 推荐：使用动词短语或描述性名词
const stop = effect(() => {
  console.log('Count:', count.get());
});

effect(() => {
  document.title = `Count: ${count.get()}`;
});

// 清理函数命名
const unsubscribe = effect(() => {
  const id = setup();
  return () => cleanup(id);
});

// ❌ 不推荐
const e = effect(() => ...);          // 无意义
const eff = effect(() => ...);        // 缩写
const run = effect(() => ...);        // 不够具体
```

### Link 和 Node 相关

```typescript
// ✅ 推荐
const link: Link = { version, dep, sub, ... };
const node: ReactiveNode = { deps, subs, flags };
const currentDep = node.deps;
const nextLink = link.nextDep;

// ❌ 不推荐
const l: Link = ...;                  // 单字母
const lnk: Link = ...;                // 缩写
const currentNodeInList: Link = ...;  // 冗长
```

## 🎯 类型和泛型命名

### 泛型参数

```typescript
// ✅ 推荐：使用单个大写字母
export class Signal<T = unknown> { }
export function computed<T>(fn: () => T): Computed<T> { }

// 多个泛型时使用有意义的名称
export class Computed<T = any, S = T> { 
  // T: 当前值类型
  // S: setter 参数类型（默认为 T）
}

// ❌ 不推荐
export class Signal<Type> { }           // 过长
export class Signal<t> { }              // 小写
export class Computed<T, U> { }         // 无意义
```

### 类型别名

```typescript
// ✅ 推荐：使用 PascalCase
type SignalGetter<T> = () => T;
type EffectCleanup = () => void;
type ReactiveFlags = number;

// ❌ 不推荐
type signalGetter<T> = () => T;         // 小写开头
type SIGNAL_GETTER<T> = () => T;        // 全大写
type getter<T> = () => T;               // 小写
```

### 接口扩展

```typescript
// ✅ 推荐：清晰表达继承关系
interface ReactiveNode {
  deps?: Link;
  flags: ReactiveFlags;
}

interface SignalNode<T> extends ReactiveNode {
  currentValue?: T;
  pendingValue?: T;
}

interface ComputedNode<T, S> extends SignalNode<T> {
  getter: (previousValue?: T) => T;
  setter?: (newValue: S) => void;
}

// ❌ 不推荐
interface SignalNodeExtended extends ReactiveNode { } // 不清晰
interface SignalNode2 extends ReactiveNode { }        // 数字后缀
```

## 🔄 动作和方法命名

### Getter/Setter 方法

```typescript
// ✅ 推荐
class Signal<T> {
  get(): T;           // 获取值
  set(value: T): void; // 设置值
}

class Computed<T> {
  get(): T;           // 获取计算值
  peek(): T;          // 获取值但不追踪依赖
}

// ❌ 不推荐
class Signal<T> {
  getValue(): T;      // 冗余
  setValue(v: T): void; // 冗余
  read(): T;          // 不标准
  write(v: T): void;  // 不标准
}
```

### 生命周期方法

```typescript
// ✅ 推荐
function startBatch(): void;      // 开始批量更新
function endBatch(): void;        // 结束批量更新
function flush(): void;           // 执行待处理更新
function dispose(): void;         // 销毁资源
function cleanup(): void;         // 清理副作用

// ❌ 不推荐
function beginBatch(): void;      // 不一致
function finishBatch(): void;     // 不一致
function runFlush(): void;        // 冗余
function destroy(): void;         // 过于激进
```

### 依赖管理方法

```typescript
// ✅ 推荐
function link(dep: ReactiveNode, sub: ReactiveNode): void;      // 建立依赖
function unlink(node: ReactiveNode): void;                       // 移除依赖
function track(node: ReactiveNode): void;                        // 追踪依赖
function notify(node: ReactiveNode): void;                       // 通知订阅者

// ❌ 不推荐
function addDependency(): void;     // 冗长
function removeDependency(): void;  // 冗长
function dependOn(): void;          // 不清晰
```

## 📊 布尔值命名

### 状态标志

```typescript
// ✅ 推荐：使用 is/has/should/can 等前缀
const isActive = signal(true);
const hasChanged = value !== oldValue;
const shouldUpdate = flags & ReactiveFlags.Dirty;
const canCompute = deps !== undefined;

// 标志位命名
const enum ReactiveFlags {
  IS_DIRTY = 1,      // 需要更新
  IS_WATCHING = 2,   // 正在监听
  CAN_MUTATE = 4,    // 可以变异
}

// ❌ 不推荐
const active = signal(true);        // 不明确
const dirty = true;                 // 形容词，不是完整的判断
const watch = false;                // 动词，不是状态
```

### 条件判断

```typescript
// ✅ 推荐
if (node.flags & ReactiveFlags.Dirty) {
  // 需要更新
}

if (!shouldSkip) {
  // 不应该跳过
}

// ❌ 不推荐
if (dirty) {             // 不完整
if (flag === 16) {       // 魔术数字
if (skip === false) {    // 双重否定
```

## 📈 命名一致性检查清单

在提交代码前，请确认：

- [ ] 变量使用 camelCase
- [ ] 类/接口使用 PascalCase
- [ ] 常量使用 UPPER_SNAKE_CASE
- [ ] 文件使用 kebab-case
- [ ] 泛型使用单个大写字母
- [ ] 布尔值使用 is/has/should 前缀
- [ ] 方法使用动词开头
- [ ] 命名语义清晰
- [ ] 无单字母变量（除循环计数器外）
- [ ] 无模糊缩写
