## 🎯 核心 API 测试模板

### 1. Signal 测试

```typescript
import { describe, test, expect } from 'vitest';
import { signal, effect } from '../src';

describe('signal', () => {
  describe('创建', () => {
    test('应该创建无初始值的信号', () => {
      const sig = signal<number>();
      expect(sig.get()).toBeUndefined();
    });
    
    test('应该创建有初始值的信号', () => {
      const count = signal(0);
      expect(count.get()).toBe(0);
    });
    
    test('应该支持 null 作为初始值', () => {
      const nullable = signal<string | null>(null);
      expect(nullable.get()).toBeNull();
    });
  });
  
  describe('get()', () => {
    test('应该返回当前值', () => {
      const count = signal(5);
      expect(count.get()).toBe(5);
    });
    
    test('应该在 activeSub 存在时建立依赖', () => {
      const count = signal(0);
      let effectRunCount = 0;
      
      effect(() => {
        effectRunCount++;
        count.get(); // 建立依赖
      });
      
      expect(effectRunCount).toBe(1);
      
      count.set(1);
      expect(effectRunCount).toBe(2); // 依赖触发
    });
  });
  
  describe('set()', () => {
    test('应该更新值', () => {
      const count = signal(0);
      count.set(10);
      expect(count.get()).toBe(10);
    });
    
    test('相同值不应该触发更新', () => {
      const count = signal(0);
      let callCount = 0;
      
      effect(() => {
        callCount++;
        count.get();
      });
      
      expect(callCount).toBe(1);
      count.set(0); // 相同值
      expect(callCount).toBe(1); // 不会重新执行
    });
    
    test('应该触发依赖的 effect', () => {
      const count = signal(0);
      let effectValue: number;
      
      effect(() => {
        effectValue = count.get() * 2;
      });
      
      expect(effectValue).toBe(0);
      
      count.set(5);
      expect(effectValue).toBe(10);
    });
  });
});
```

### 2. Computed 测试

```typescript
import { describe, test, expect } from 'vitest';
import { signal, computed, effect } from '../src';

describe('computed', () => {
  describe('基础计算', () => {
    test('应该计算并缓存结果', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      
      expect(doubled.get()).toBe(10);
      expect(doubled.get()).toBe(10); // 缓存命中
    });
    
    test('应该在依赖变化时重新计算', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      
      expect(doubled.get()).toBe(10);
      
      count.set(10);
      expect(doubled.get()).toBe(20);
    });
    
    test('应该支持嵌套 computed', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      const quadrupled = computed(() => doubled.get() * 2);
      
      expect(quadrupled.get()).toBe(20);
      
      count.set(10);
      expect(quadrupled.get()).toBe(40);
    });
  });
  
  describe('惰性求值', () => {
    test('应该只在访问时计算', () => {
      const count = signal(5);
      let computeCount = 0;
      
      computed(() => {
        computeCount++;
        return count.get() * 2;
      });
      
      expect(computeCount).toBe(0); // 未访问，未计算
      
      const value = doubled.get();
      expect(computeCount).toBe(1);
      expect(value).toBe(10);
    });
  });
});
```

### 3. Effect 测试

```typescript
import { describe, test, expect, vi } from 'vitest';
import { signal, effect } from '../src';

describe('effect', () => {
  test('应该立即执行一次', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
  });
  
  test('应该在依赖变化时重新执行', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
    
    count.set(1);
    expect(spy).toHaveBeenCalledTimes(2);
    
    count.set(2);
    expect(spy).toHaveBeenCalledTimes(3);
  });
  
  test('应该返回清理函数', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    const stop = effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
    
    stop(); // 停止监听
    
    count.set(1);
    expect(spy).toHaveBeenCalledTimes(1); // 不再执行
  });
});
```

## 🎯 批量更新测试

```typescript
import { describe, test, expect } from 'vitest';
import { signal, computed, startBatch, endBatch } from '../src';

describe('批量更新', () => {
  test('应该合并多次更新', () => {
    const a = signal(0);
    const b = signal(0);
    const sum = computed(() => a.get() + b.get());
    
    let updateCount = 0;
    effect(() => {
      updateCount++;
      sum.get();
    });
    
    expect(updateCount).toBe(1);
    
    // 批量更新
    startBatch();
    a.set(1);
    b.set(2);
    endBatch();
    
    expect(sum.get()).toBe(3);
    expect(updateCount).toBe(2); // 只触发一次更新
  });
  
  test('应该支持嵌套批量更新', () => {
    const a = signal(0);
    
    startBatch();
    a.set(1);
    
    startBatch();
    a.set(2);
    endBatch();
    
    a.set(3);
    endBatch();
    
    expect(a.get()).toBe(3);
  });
});
```

## 🔍 边界情况测试

### 循环依赖检测

```typescript
test('应该检测循环依赖', () => {
  const a = signal(0);
  
  // 创建可能导致循环的场景
  const b = computed(() => a.get() + 1);
  
  expect(() => {
    // 这里应该不会导致无限循环
    b.get();
  }).not.toThrow();
});
```

### 内存泄漏检测

```typescript
test('应该清理 effect 避免内存泄漏', () => {
  const count = signal(0);
  let effectRan = false;
  
  const stop = effect(() => {
    effectRan = true;
    count.get();
  });
  
  expect(effectRan).toBe(true);
  
  stop();
  
  // 清理后，依赖应该被移除
  expect(count.subs).toBeUndefined();
});
```
