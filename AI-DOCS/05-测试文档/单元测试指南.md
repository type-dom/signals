## 🛠️ Vitest 基础

### 安装和配置

```bash
npm install -D vitest @vitest/coverage-v8
```

```typescript
// vitest.config.ts
import { defineConfig } from 'vitest/config';

export default defineConfig({
  test: {
    globals: true,  // 全局注入 describe, it, expect
    environment: 'node',
    coverage: {
      provider: 'v8',
      reporter: ['text', 'json', 'html'],
    },
  },
});
```

### 运行测试

```bash
# 运行所有测试
npx vitest

# 运行特定文件
npx vitest tests/signal.test.ts

# 运行匹配名称的测试
npx vitest -t "should create"

# 生成覆盖率报告
npx vitest --coverage

# 监听模式
npx vitest --watch
```

## 📝 测试结构

### 基础模板

```typescript
import { describe, test, expect, beforeEach, afterEach } from 'vitest';
import { signal } from '../src';

describe('signal', () => {
  // 测试套件
  
  beforeEach(() => {
    // 每个测试前执行
  });
  
  afterEach(() => {
    // 每个测试后执行
  });
  
  test('应该创建工作', () => {
    // 单个测试用例
    const count = signal(0);
    expect(count.get()).toBe(0);
  });
});
```

### 描述性命名

```typescript
// ✅ 推荐：清晰的描述
test('should create signal with initial value');
test('should trigger effect when value changes');
test('should not trigger if value is same');

// ❌ 不推荐：模糊的命名
test('basic test');
test('test 1');
test('update test');
```

## 🎯 核心 API 测试

### Signal 测试

```typescript
describe('signal', () => {
  describe('创建', () => {
    test('应该创建无初始值的信号', () => {
      const sig = signal<number>();
      expect(sig.get()).toBeUndefined();
    });
    
    test('应该创建有初始值的信号', () => {
      const count = signal(0);
      expect(count.get()).toBe(0);
    });
    
    test('应该支持 null 作为初始值', () => {
      const nullable = signal<string | null>(null);
      expect(nullable.get()).toBeNull();
    });
  });
  
  describe('get()', () => {
    test('应该返回当前值', () => {
      const count = signal(5);
      expect(count.get()).toBe(5);
    });
    
    test('应该追踪依赖', () => {
      const count = signal(0);
      let runCount = 0;
      
      effect(() => {
        runCount++;
        count.get();
      });
      
      expect(runCount).toBe(1);
      count.set(1);
      expect(runCount).toBe(2);
    });
  });
  
  describe('set()', () => {
    test('应该更新值', () => {
      const count = signal(0);
      count.set(10);
      expect(count.get()).toBe(10);
    });
    
    test('相同值不应该触发更新', () => {
      const count = signal(0);
      let callCount = 0;
      
      effect(() => {
        callCount++;
        count.get();
      });
      
      expect(callCount).toBe(1);
      count.set(0);
      expect(callCount).toBe(1);
    });
    
    test('应该触发 effect', () => {
      const count = signal(0);
      let effectValue: number;
      
      effect(() => {
        effectValue = count.get();
      });
      
      expect(effectValue!).toBe(0);
      count.set(5);
      expect(effectValue!).toBe(5);
    });
  });
});
```

### Computed 测试

```typescript
describe('computed', () => {
  describe('基础计算', () => {
    test('应该计算并缓存结果', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      
      expect(doubled.get()).toBe(10);
      expect(doubled.get()).toBe(10); // 缓存命中
    });
    
    test('应该在依赖变化时重新计算', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      
      expect(doubled.get()).toBe(10);
      
      count.set(10);
      expect(doubled.get()).toBe(20);
    });
    
    test('应该支持嵌套 computed', () => {
      const count = signal(5);
      const doubled = computed(() => count.get() * 2);
      const quadrupled = computed(() => doubled.get() * 2);
      
      expect(quadrupled.get()).toBe(20);
      
      count.set(10);
      expect(quadrupled.get()).toBe(40);
    });
  });
  
  describe('惰性求值', () => {
    test('应该只在访问时计算', () => {
      const count = signal(5);
      let computeCount = 0;
      
      const doubled = computed(() => {
        computeCount++;
        return count.get() * 2;
      });
      
      expect(computeCount).toBe(0); // 未访问
      
      doubled.get();
      expect(computeCount).toBe(1);
      
      doubled.get();
      expect(computeCount).toBe(1); // 缓存命中
    });
  });
});
```

### Effect 测试

```typescript
describe('effect', () => {
  test('应该立即执行一次', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
  });
  
  test('应该在依赖变化时重新执行', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
    
    count.set(1);
    expect(spy).toHaveBeenCalledTimes(2);
    
    count.set(2);
    expect(spy).toHaveBeenCalledTimes(3);
  });
  
  test('应该返回清理函数', () => {
    const count = signal(0);
    const spy = vi.fn();
    
    const stop = effect(() => {
      spy();
      count.get();
    });
    
    expect(spy).toHaveBeenCalledTimes(1);
    
    stop();
    
    count.set(1);
    expect(spy).toHaveBeenCalledTimes(1); // 不再执行
  });
});
```

## 🔧 Mock 和 Spy

### Mock 函数

```typescript
test('should call callback', () => {
  const callback = vi.fn();
  
  effect(() => {
    callback(count.get());
  });
  
  expect(callback).toHaveBeenCalledWith(0);
  
  count.set(5);
  expect(callback).toHaveBeenCalledWith(5);
});
```

### Spy 方法

```typescript
test('should call internal method', () => {
  const obj = {
    method() { return 'original'; }
  };
  
  const spy = vi.spyOn(obj, 'method');
  
  obj.method();
  
  expect(spy).toHaveBeenCalled();
  
  spy.mockRestore();
});
```

### Mock 定时器

```typescript
vi.useFakeTimers();

test('should handle timers', () => {
  const callback = vi.fn();
  
  setInterval(() => {
    callback(count.get());
  }, 1000);
  
  vi.advanceTimersByTime(3000);
  
  expect(callback).toHaveBeenCalledTimes(3);
});

vi.useRealTimers();
```

## ⚡ 异步测试

### Async/Await

```typescript
test('should handle async', async () => {
  const data = signal<string>('');
  
  await fetchData().then(result => {
    data.set(result);
  });
  
  expect(data.get()).toBe('expected');
});
```

### Promise

```typescript
test('should resolve promise', () => {
  return fetchData().then(result => {
    expect(result).toBe('expected');
  });
});
```

### Resolves/Rejects

```typescript
test('should resolve', async () => {
  await expect(Promise.resolve('value')).resolves.toBe('value');
});

test('should reject', async () => {
  await expect(Promise.reject('error')).rejects.toBe('error');
});
```

## 🎯 边界条件测试

### 空值处理

```typescript
test('should handle null', () => {
  const nullable = signal<string | null>(null);
  expect(nullable.get()).toBeNull();
  
  nullable.set('value');
  expect(nullable.get()).toBe('value');
  
  nullable.set(null);
  expect(nullable.get()).toBeNull();
});
```

### 对象引用

```typescript
test('should handle object references', () => {
  const obj = signal({ name: 'John' });
  
  const ref = obj.get();
  ref.name = 'Jane';
  
  // Signal 中的对象未被修改（如果是浅层）
  expect(obj.get().name).toBe('John');
});
```

### 数组操作

```typescript
test('should handle array mutations', () => {
  const items = signal<number[]>([1, 2, 3]);
  
  // 替换整个数组
  items.set([4, 5, 6]);
  expect(items.get()).toEqual([4, 5, 6]);
});
```
