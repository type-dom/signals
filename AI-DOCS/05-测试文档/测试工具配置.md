## 🔧 Vitest 完整配置

### 基础配置

```typescript
// vitest.config.ts
import { defineConfig } from 'vitest/config';
import path from 'path';

export default defineConfig({
  // Vite 配置
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
  
  // Vitest配置
  test: {
    // 全局变量
    globals: true,
    
    // 测试环境
    environment: 'node',
    
    // 线程池
    pool: 'threads',
    poolOptions: {
      threads: {
        minThreads: 2,
        maxThreads: 4,
      },
    },
    
    // 超时设置
    testTimeout: 30000,
    hookTimeout: 30000,
    
    // 设置文件
    setupFiles: ['./tests/setup.ts'],
    
    // 全局测试匹配
    include: ['tests/**/*.test.ts'],
    exclude: ['tests/**/*.skip.test.ts'],
    
    // 覆盖率配置
    coverage: {
      provider: 'v8',
      reporter: ['text', 'json', 'html', 'lcov'],
      reportsDirectory: './coverage',
      
      // 阈值配置
      threshold: {
        global: {
          branches: 90,
          functions: 90,
          lines: 90,
          statements: 90,
        },
      },
      
      // 包含/排除的文件
      include: ['src/**/*.ts'],
      exclude: [
        'src/index.ts',
        '**/*.d.ts',
        '**/node_modules/**',
      ],
    },
    
    // 报告器
    reporters: ['default', 'html'],
    
    // 输出目录
    outputFile: {
      html: 'test-report.html',
      json: 'test-report.json',
    },
  },
});
```

### 环境配置

```typescript
// tests/setup.ts
import { beforeAll, afterAll } from 'vitest';

// 全局设置
beforeAll(() => {
  // 测试前执行一次
  console.log('Starting tests...');
});

afterAll(() => {
  // 测试后执行一次
  console.log('Tests completed.');
});

// Mock console.error 以避免测试污染
const originalError = console.error;
beforeAll(() => {
  console.error = (...args: any[]) => {
    if (args[0]?.includes?.('Warning')) return;
    originalError.call(console, ...args);
  };
});

afterAll(() => {
  console.error = originalError;
});
```

## 🌐 Playwright 配置

### 安装

```bash
npm install -D @playwright/test
npx playwright install
```

### 配置文件

```typescript
// playwright.config.ts
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  // 测试目录
  testDir: './tests/e2e',
  
  // 超时设置
  timeout: 30 * 1000,
  expect: {
    timeout: 5000,
  },
  
  // 失败重试
  retries: process.env.CI ? 2 : 0,
  
  // 并行执行
  workers: process.env.CI ? 1 : undefined,
  
  // 报告器
  reporter: 'html',
  
  // 共享配置
  use: {
    baseURL: 'http://localhost:3000',
    
    // 截图
    screenshot: 'only-on-failure',
    
    // 录屏
    video: 'retain-on-failure',
    
    // 追踪
    trace: 'retain-on-failure',
  },
  
  // 浏览器配置
  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
    {
      name: 'firefox',
      use: { ...devices['Desktop Firefox'] },
    },
    {
      name: 'webkit',
      use: { ...devices['Desktop Safari'] },
    },
    {
      name: 'Mobile Chrome',
      use: { ...devices['Pixel 5'] },
    },
    {
      name: 'Mobile Safari',
      use: { ...devices['iPhone 12'] },
    },
  ],
  
  // 启动服务器
  webServer: {
    command: 'npm run dev',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI,
  },
});
```

### 测试示例

```typescript
// tests/e2e/reactivity.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Reactivity System', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/demo');
  });
  
  test('should update on signal change', async ({ page }) => {
    await expect(page.locator('#count')).toHaveText('0');
    
    await page.click('button.increment');
    
    await expect(page.locator('#count')).toHaveText('1');
  });
  
  test('should handle computed', async ({ page }) => {
    await page.click('button.increment');
    await page.click('button.increment');
    
    await expect(page.locator('#doubled')).toHaveText('4');
  });
});
```

## 🧪 Testing Library 配置

### 安装

```bash
npm install -D @testing-library/dom @testing-library/jest-dom
```

### 配置

```typescript
// tests/setup.ts
import '@testing-library/jest-dom/vitest';

import { cleanup } from '@testing-library/dom';
import { afterEach } from 'vitest';

// 每个测试后清理 DOM
afterEach(() => {
  cleanup();
});
```

### 使用示例

```typescript
import { render, screen, fireEvent } from '@testing-library/dom';
import { describe, test, expect } from 'vitest';

describe('Component', () => {
  test('should render correctly', () => {
    render(<div data-testid="test">Hello</div>);
    
    expect(screen.getByTestId('test')).toBeInTheDocument();
    expect(screen.getByText('Hello')).toBeInTheDocument();
  });
  
  test('should handle events', () => {
    const handleClick = vi.fn();
    
    render(<button onClick={handleClick}>Click</button>);
    
    fireEvent.click(screen.getByRole('button'));
    
    expect(handleClick).toHaveBeenCalledTimes(1);
  });
});
```

## 🎭 Mock 工具配置

### Vi Mock

```typescript
import { vi } from 'vitest';

// Mock 函数
const mockFn = vi.fn((x) => x * 2);
expect(mockFn(5)).toBe(10);

// Mock 实现
mockFn.mockImplementation(x => x + 1);
expect(mockFn(5)).toBe(6);

// Mock 一次
mockFn.mockReturnValueOnce(42);
expect(mockFn(1)).toBe(42);
expect(mockFn(1)).toBe(2); // 恢复正常

// 异步 Mock
mockFn.mockResolvedValue({ data: 'test' });
await expect(mockFn()).resolves.toEqual({ data: 'test' });

// Mock 拒绝
mockFn.mockRejectedValue(new Error('error'));
await expect(mockFn()).rejects.toThrow('error');
```

### Mock 模块

```typescript
// 完全 Mock
vi.mock('../api/user', () => ({
  getUser: vi.fn(),
  createUser: vi.fn(),
  deleteUser: vi.fn(),
}));

// 部分 Mock
vi.mock('../utils', async () => {
  const actual = await vi.importActual('../utils');
  return {
    ...actual,
    heavyFunction: vi.fn(),
  };
});

// Mock 第三方库
vi.mock('axios', () => ({
  default: {
    get: vi.fn(),
    post: vi.fn(),
  },
}));
```

### Mock 定时器

```typescript
vi.useFakeTimers();

test('should handle timers', () => {
  const callback = vi.fn();
  
  setInterval(() => {
    callback();
  }, 1000);
  
  // 快进时间
  vi.advanceTimersByTime(3000);
  
  expect(callback).toHaveBeenCalledTimes(3);
});

test('should handle async timers', async () => {
  vi.useFakeTimers();
  
  setTimeout(() => {
    console.log('After 1s');
  }, 1000);
  
  await vi.runAllTimersAsync();
});

// 恢复真实定时器
vi.useRealTimers();
```


## 🚀 CI/CD 集成

### GitHub Actions

```yaml
# .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run unit tests
        run: npm run test:unit
      
      - name: Run E2E tests
        run: npx playwright test
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info
          flags: unittests
```

### GitLab CI

```yaml
# .gitlab-ci.yml
test:
  image: node:18
  script:
    - npm ci
    - npm run test:unit
    - npm run test:e2e
  artifacts:
    reports:
      coverage_report:
        coverage_format: cobertura
        path: coverage/cobertura-coverage.xml
```
