## 📝 单元测试规范

### 测试范围

**必须测试的内容**:
- ✅ 所有 public API
- ✅ 边界条件和异常处理
- ✅ 依赖追踪逻辑
- ✅ 更新传播机制
- ✅ 内存泄漏检测

**可以不测试的内容**:
- ⚪ 私有实现细节
- ⚪ 类型定义（由 TypeScript 保证）
- ⚪ 第三方库功能

### 测试结构

```typescript
import { describe, test, expect, beforeEach, vi } from 'vitest';
import { signal, computed, effect } from '../src';

describe('signal', () => {
  describe('Initialization', () => {
    test('should create with undefined value', () => {
      const sig = signal<number>();
      expect(sig.get()).toBeUndefined();
    });
    
    test('should create with initial value', () => {
      const count = signal(0);
      expect(count.get()).toBe(0);
    });
  });
  
  describe('Get operations', () => {
    test('should return current value', () => {
      const count = signal(5);
      expect(count.get()).toBe(5);
    });
    
    test('should track dependencies', () => {
      const count = signal(0);
      let effectRunCount = 0;
      
      effect(() => {
        effectRunCount++;
        count.get();
      });
      
      expect(effectRunCount).toBe(1);
      count.set(1);
      expect(effectRunCount).toBe(2);
    });
  });
  
  describe('Set operations', () => {
    test('should update value', () => {
      const count = signal(0);
      count.set(10);
      expect(count.get()).toBe(10);
    });
    
    test('should not trigger for same value', () => {
      const count = signal(0);
      let callCount = 0;
      
      effect(() => {
        callCount++;
        count.get();
      });
      
      expect(callCount).toBe(1);
      count.set(0); // 相同值
      expect(callCount).toBe(1);
    });
  });
});
```

## 🔗 集成测试规范

### 测试场景

**典型集成测试**:
- ✅ Signal → Computed → Effect 链路
- ✅ 批量更新流程
- ✅ 作用域管理
- ✅ 循环依赖检测

### 示例

```typescript
describe('Integration: Signal Chain', () => {
  test('should propagate updates through chain', () => {
    const count = signal(0);
    const doubled = computed(() => count.get() * 2);
    const quadrupled = computed(() => doubled.get() * 2);
    
    let result: number;
    
    effect(() => {
      result = quadrupled.get();
    });
    
    expect(result!).toBe(0);
    
    count.set(5);
    expect(result!).toBe(20);
  });
  
  test('should handle batch updates', () => {
    const a = signal(0);
    const b = signal(0);
    const sum = computed(() => a.get() + b.get());
    
    let updateCount = 0;
    
    effect(() => {
      updateCount++;
      sum.get();
    });
    
    expect(updateCount).toBe(1);
    
    startBatch();
    a.set(1);
    b.set(2);
    endBatch();
    
    expect(sum.get()).toBe(3);
    expect(updateCount).toBe(2); // 只触发一次
  });
});
```

## 🌐 E2E 测试规范

### 测试工具

使用 Playwright 进行端到端测试：

```typescript
// tests/e2e/reactivity.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Reactivity System', () => {
  test('should update UI on signal change', async ({ page }) => {
    await page.goto('/demo');
    
    // 初始状态
    await expect(page.locator('#count')).toHaveText('0');
    
    // 点击按钮
    await page.click('button.increment');
    
    // 验证更新
    await expect(page.locator('#count')).toHaveText('1');
    await expect(page.locator('#doubled')).toHaveText('2');
  });
  
  test('should handle rapid updates', async ({ page }) => {
    await page.goto('/demo');
    
    // 快速点击 10 次
    for (let i = 0; i < 10; i++) {
      await page.click('button.increment');
    }
    
    // 验证最终状态
    await expect(page.locator('#count')).toHaveText('10');
  });
});
```

## 🧪 Mock 与 Stub 规范

### Mock 函数

```typescript
import { vi } from 'vitest';

// 创建 mock 函数
const mockFn = vi.fn();

// 调用
mockFn('arg1', 'arg2');

// 断言
expect(mockFn).toHaveBeenCalled();
expect(mockFn).toHaveBeenCalledWith('arg1', 'arg2');
expect(mockFn).toHaveBeenCalledTimes(3);
```

### Mock 模块

```typescript
import { vi } from 'vitest';

// Mock 整个模块
vi.mock('../api', () => ({
  fetchData: vi.fn(() => Promise.resolve({ data: [] })),
}));

// 部分 Mock
vi.mock('../utils', async () => {
  const actual = await vi.importActual('../utils');
  return {
    ...actual,
    heavyComputation: vi.fn(() => 42),
  };
});
```

### Spy 方法

```typescript
import { vi } from 'vitest';

const obj = {
  method() { return 'original'; }
};

// Spy 方法
const spy = vi.spyOn(obj, 'method');

// 调用
obj.method();

// 断言
expect(spy).toHaveBeenCalled();
expect(spy).toReturnWith('original');

// 恢复
spy.mockRestore();
```

## ⚡ 性能测试

### 基准测试

```typescript
import { bench, describe } from 'vitest';

describe('Performance Benchmarks', () => {
  bench('signal creation', () => {
    const sig = signal(0);
  });
  
  bench('signal get/set', () => {
    const sig = signal(0);
    sig.set(sig.get() + 1);
  });
  
  bench('computed chain', () => {
    const a = signal(0);
    const b = computed(() => a.get() + 1);
    const c = computed(() => b.get() + 1);
    c.get();
  });
  
  bench('effect execution', () => {
    const sig = signal(0);
    effect(() => {
      sig.get();
    });
    sig.set(sig.get() + 1);
  });
});
```

### 内存泄漏检测

```typescript
import { WeakRef } from 'weakref';

test('should not leak memory', async () => {
  let ref: WeakRef<any>;
  
  {
    const sig = signal({ large: new Array(1000000) });
    ref = new WeakRef(sig);
  }
  
  // 强制 GC
  global.gc?.();
  
  // 等待清理
  await new Promise(resolve => setTimeout(resolve, 100));
  
  // 验证是否被回收
  expect(ref.deref()).toBeUndefined();
});
```

## 📈 测试质量指标

### 好的测试特征

- ✅ **可重复性** - 每次运行结果一致
- ✅ **独立性** - 测试之间无依赖
- ✅ **快速执行** - 单个测试 < 100ms
- ✅ **易于理解** - 测试意图清晰
- ✅ **全面覆盖** - 正常 + 异常场景

### 测试异味（需要重构）

- ❌ 测试过于复杂（> 50 行）
- ❌ 依赖其他测试的执行顺序
- ❌ 使用 sleep/wait 等待异步
- ❌ 断言过多（> 5 个）
- ❌ 测试名称不清晰
